<script>
(function(){
  const carousel = document.querySelector('.talk-carousel');
  if (!carousel) return;
  const track = carousel.querySelector('.talk-slides');
  const slides = carousel.querySelectorAll('.talk-slide');
  const prev = carousel.querySelector('.prev');
  const next = carousel.querySelector('.next');
  const dots = document.querySelectorAll('.talk-dot');
  let index = 0;

  function update() {
    track.style.transform = 'translateX(' + (-index * 100) + '%)';
    dots.forEach((d,i)=>{ d.classList.toggle('active', i===index); d.setAttribute('aria-selected', i===index); });
  }
  prev.addEventListener('click', ()=>{ index = (index -1 + slides.length) % slides.length; update(); });
  next.addEventListener('click', ()=>{ index = (index +1) % slides.length; update(); });
  dots.forEach(d => d.addEventListener('click', e => { index = Number(e.currentTarget.dataset.index); update(); }));
  carousel.addEventListener('keydown', e => { if (e.key==='ArrowLeft') prev.click(); if (e.key==='ArrowRight') next.click(); });
  update();
})();
</script>

This talk presents the design and evaluation of instruction methods for DailyBuddy, a mobile app to support adults with autism in developing daily living skills. We compare guidance strategies (step-by-step, contextual prompts, and multimedia cues), report usability and learning outcomes, and distill practical design recommendations for accessible mobile instruction.
